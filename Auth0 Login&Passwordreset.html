<script>

let isSignupMode = false;


function togglePassword() {

const input = document.getElementById("password");

const span = event.target;

if (input.type === "password") {

input.type = "text";

span.textContent = "Hide";

} else {

input.type = "password";

span.textContent = "Show";

}

}


function displayError(message) {

const errorDiv = document.getElementById("error-message");

errorDiv.innerText = message;

errorDiv.classList.remove("hidden");

}


function updateAuthModeUI() {

const forgotPass = document.getElementById("forgot-password-link");

const heading = document.querySelector("h2");

const submitBtn = document.getElementById("btn-login");

const toggleLink = document.getElementById("toggle-auth-mode");

const toggleText = document.getElementById("toggle-text")


if (isSignupMode) {

heading.textContent = "Sign up";

submitBtn.textContent = "Sign Up";

toggleLink.textContent = "Login";

toggleText.textContent = "Already have an account?";

forgotPass.classList.add('hidden')

} else {

heading.textContent = "Sign in";

submitBtn.textContent = "Sign In";

toggleLink.textContent = "Sign up";

toggleText.textContent = "Don't have an account?";

forgotPass.classList.remove('hidden')


}

}


document.addEventListener("DOMContentLoaded", () => {

document.getElementById("language-select").addEventListener("change", function () {

alert("Language changed to: " + this.value);

});


const passwordInput = document.getElementById("password");

const toggleSpan = document.getElementById("toggle-password");


passwordInput.addEventListener("input", function () {

if (this.value.trim() === "") {

toggleSpan.classList.add("hidden");

this.type = "password";

toggleSpan.textContent = "Show";

} else {

toggleSpan.classList.remove("hidden");

}

});


document.getElementById("toggle-auth-mode").addEventListener("click", () => {

isSignupMode = !isSignupMode;

updateAuthModeUI();

});

});


window.addEventListener("load", function () {

const config = JSON.parse(decodeURIComponent(escape(window.atob('@@config@@'))));

const leeway = parseInt(config.internalOptions.leeway || 0);

if (!isNaN(leeway)) config.internalOptions.leeway = leeway;


const webAuth = new auth0.WebAuth({

domain: config.auth0Domain,

clientID: config.clientID,

redirectUri: config.callbackURL,

responseType: "code",

scope: config.internalOptions.scope,

_csrf: config.internalOptions._csrf,

state: config.internalOptions.state,

_intstate: config.internalOptions._intstate,

overrides: {

__tenant: config.auth0Tenant,

__token_issuer: config.authorizationServer.issuer,

},

});


const databaseConnection = "Username-Password-Authentication";




document.getElementById("forgot-password-link").addEventListener("click", function (e) {

e.preventDefault();

const email = document.getElementById("email").value.trim();

if (!email) {

displayError("Please enter your email address to reset your password.");

return;

}


webAuth.changePassword({

connection: 'Username-Password-Authentication',

email: email

}, function (err, resp) {

if (err) {

displayError(err.description || err.message || "Error sending reset email.");

} else {

alert("Password reset email sent. Please check your inbox.");

}

});

});


document.getElementById("btn-login").addEventListener("click", function (e) {

e.preventDefault();


const email = document.getElementById("email").value.trim();

const password = document.getElementById("password").value;


if (!email || !password) {

displayError("Please enter both email and password.");

return;

}


if (isSignupMode) {

webAuth.signup({

connection: databaseConnection,

email: email,

password: password,

}, function (err) {

if (err) {

displayError(err.description || err.message || "Signup failed.");

return;

}


// Auto login after successful signup

webAuth.login({

realm: databaseConnection,

username: email,

password: password,

}, function (err) {

if (err) {

displayError(err.description || err.message || "Login after signup failed.");

}

});

});

} else {

webAuth.login({

realm: databaseConnection,

username: email,

password: password,

}, function (err) {

if (err) {

displayError(err.description || err.message || "Login failed.");

}

});

}

});


document.getElementById("btn-google").addEventListener("click", function () {

webAuth.authorize({

connection: "google-oauth2",

}, function (err) {

if (err) displayError(err.description || err.message || "Login failed.");

});

});

});

</script>

<script>

var resetView = $("#reset-view");

var completedView = $("#completed-view");

var errorView = $("#error-view");

var form = $("#change-password-form");


form.submit(function (e) {

e.preventDefault();


$.ajax({

type: form.attr("method"),

url: form.attr("action"),

data: form.serialize(),

success: handleSuccess,

error: handleError

});

});


var handleSuccess = function (res) {

if (res.body && typeof res.body.result_url === "string") {

return window.location.replace(res.body.result_url);

} else {

resetView.hide();

errorView.hide();

completedView.show();

}

};


var handleError = function (res) {

var getErrorFunc = !!res ? getResponseError : getNetworkError;

var error = getErrorFunc(res);

completedView.hide();

resetView.show();

errorView.show().text(error);

};


var getResponseError = function (res) {

var body = res.body || {};

var passwordErrors = {

PasswordStrengthError: "Password is too weak.",

PasswordHistoryError: "You’ve used this password before.",

PasswordDictionaryError: "Avoid common dictionary words.",

PasswordNoUserInfoError: "Password shouldn't contain your email or name."

};

return passwordErrors[body.name] || "An unexpected server error occurred.";

};


var getNetworkError = function () {

return "A network error occurred. Please try again.";

};

</script>

<script>

function togglePassword(inputId, toggleId) {

const input = document.getElementById(inputId);

const toggle = document.getElementById(toggleId);

if (input.type === "password" && input.type!=="") {

input.type = "text";

toggle.textContent = "Hide";

} else {

input.type = "password";

toggle.textContent = "Show";

}

}

function toggleShow(inputId, toggleId) {

const input = document.getElementById(inputId);

const toggle = document.getElementById(toggleId);

if (input.value === "") {

toggle.classList.add('hidden')

toggle.textContent = "Show";

input.type = "password";

}

else{

toggle.classList.remove('hidden')

}}
